空のオブジェクトを作成して、このスクリプトをつける
その後、空のオブジェクトのスクリプトに、objectを指定する 

    
using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class ObjectGenerator : MonoBehaviour {
  public GameObject object;
  void Start () {
    /* なし（消しても良い） */
  }
  void Update () {
    if (Input.touchCount > 0) { //タッチした時
      GameObject obj = Instantiate (object) as GameObject;
      float px = Random.Range (-8.0f, 8.0f);
      float py = Random.Range (-8.0f, 8.0f);
      obj.transform.position = new Vector3 (px, 2.0f, 0.0f);
    }
  }
}
